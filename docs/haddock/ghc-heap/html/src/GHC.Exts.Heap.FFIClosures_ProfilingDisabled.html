-- | Get non-pointer fields from @StgTSO_@ (@TSO.h@)
peekTSOFields :: (Ptr a -> IO (Maybe CostCentreStack)) -> Ptr tsoPtr -> IO TSOFields
peekTSOFields decodeCCS ptr = do
    what_next' <- ((\hsc_ptr -> peekByteOff hsc_ptr 32)) ptr
{-# LINE 36 "GHC/Exts/Heap/FFIClosures_ProfilingDisabled.hsc" #-}
    why_blocked' <- ((\hsc_ptr -> peekByteOff hsc_ptr 34)) ptr
{-# LINE 37 "GHC/Exts/Heap/FFIClosures_ProfilingDisabled.hsc" #-}
    flags' <- ((\hsc_ptr -> peekByteOff hsc_ptr 36)) ptr
{-# LINE 38 "GHC/Exts/Heap/FFIClosures_ProfilingDisabled.hsc" #-}
    threadId' <- ((\hsc_ptr -> peekByteOff hsc_ptr 48)) ptr
{-# LINE 39 "GHC/Exts/Heap/FFIClosures_ProfilingDisabled.hsc" #-}
    saved_errno' <- ((\hsc_ptr -> peekByteOff hsc_ptr 56)) ptr
{-# LINE 40 "GHC/Exts/Heap/FFIClosures_ProfilingDisabled.hsc" #-}
    dirty' <- ((\hsc_ptr -> peekByteOff hsc_ptr 60)) ptr
{-# LINE 41 "GHC/Exts/Heap/FFIClosures_ProfilingDisabled.hsc" #-}
    alloc_limit' <- ((\hsc_ptr -> peekByteOff hsc_ptr 104)) ptr
{-# LINE 42 "GHC/Exts/Heap/FFIClosures_ProfilingDisabled.hsc" #-}
    tot_stack_size' <- ((\hsc_ptr -> peekByteOff hsc_ptr 112)) ptr
{-# LINE 43 "GHC/Exts/Heap/FFIClosures_ProfilingDisabled.hsc" #-}
    tso_prof' <- peekStgTSOProfInfo decodeCCS ptr

    return TSOFields {
        tso_what_next = parseWhatNext what_next',
        tso_why_blocked = parseWhyBlocked why_blocked',
        tso_flags = parseTsoFlags flags',
        tso_threadId = threadId',
        tso_saved_errno = saved_errno',
        tso_dirty = dirty',
        tso_alloc_limit = alloc_limit',
        tso_tot_stack_size = tot_stack_size',
        tso_prof = tso_prof'
    }

parseWhatNext :: Word16 -> WhatNext
parseWhatNext w = case w of
                    (1) -> ThreadRunGHC
{-# LINE 60 "GHC/Exts/Heap/FFIClosures_ProfilingDisabled.hsc" #-}

{-# LINE 61 "GHC/Exts/Heap/FFIClosures_ProfilingDisabled.hsc" #-}
                    (3) -> ThreadKilled
{-# LINE 62 "GHC/Exts/Heap/FFIClosures_ProfilingDisabled.hsc" #-}
                    (4) -> ThreadComplete
{-# LINE 63 "GHC/Exts/Heap/FFIClosures_ProfilingDisabled.hsc" #-}
                    _ -> WhatNextUnknownValue w

parseTsoFlags :: Word32 -> [TsoFlags]
parseTsoFlags w | isSet (2) w = TsoLocked : parseTsoFlags (unset (2) w)
{-# LINE 87 "GHC/Exts/Heap/FFIClosures_ProfilingDisabled.hsc" #-}
                | isSet (4) w = TsoBlockx : parseTsoFlags (unset (4) w)
{-# LINE 88 "GHC/Exts/Heap/FFIClosures_ProfilingDisabled.hsc" #-}
                | isSet (8) w = TsoInterruptible : parseTsoFlags (unset (8) w)
{-# LINE 89 "GHC/Exts/Heap/FFIClosures_ProfilingDisabled.hsc" #-}
                | isSet (16) w = TsoStoppedOnBreakpoint : parseTsoFlags (unset (16) w)
{-# LINE 90 "GHC/Exts/Heap/FFIClosures_ProfilingDisabled.hsc" #-}
                | isSet (64) w = TsoMarked : parseTsoFlags (unset (64) w)
{-# LINE 91 "GHC/Exts/Heap/FFIClosures_ProfilingDisabled.hsc" #-}
                | isSet (128) w = TsoSqueezed : parseTsoFlags (unset (128) w)
{-# LINE 92 "GHC/Exts/Heap/FFIClosures_ProfilingDisabled.hsc" #-}
                | isSet (256) w = TsoAllocLimit : parseTsoFlags (unset (256) w)
{-# LINE 93 "GHC/Exts/Heap/FFIClosures_ProfilingDisabled.hsc" #-}
parseTsoFlags 0 = []
parseTsoFlags w = [TsoFlagsUnknownValue w]

isSet :: Word32 -> Word32 -> Bool
isSet bitMask w = w .&. bitMask /= 0

unset :: Word32 -> Word32 -> Word32
unset bitMask w = w `xor` bitMask

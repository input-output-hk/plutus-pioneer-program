import Control.Concurrent.QSemN ( QSemN )

-- from transformers-base:
import Control.Monad.Base ( MonadBase, liftBase )

#include "inlinable.h"

--------------------------------------------------------------------------------
-- * QSemNs

-- | Generalized version of 'QSemN.newQSemN'.
newQSemN :: MonadBase IO m => Int -> m QSemN
newQSemN :: forall (m :: * -> *). MonadBase IO m => Int -> m QSemN
newQSemN = forall (b :: * -> *) (m :: * -> *) α. MonadBase b m => b α -> m α
liftBase forall b c a. (b -> c) -> (a -> b) -> a -> c
. Int -> IO QSemN
QSemN.newQSemN
{-# INLINABLE newQSemN #-}

-- | Generalized version of 'QSemN.waitQSemN'.
waitQSemN :: MonadBase IO m => QSemN -> Int -> m ()
waitQSemN :: forall (m :: * -> *). MonadBase IO m => QSemN -> Int -> m ()
waitQSemN QSemN
sem = forall (b :: * -> *) (m :: * -> *) α. MonadBase b m => b α -> m α
liftBase forall b c a. (b -> c) -> (a -> b) -> a -> c
. QSemN -> Int -> IO ()
QSemN.waitQSemN QSemN
sem
{-# INLINABLE waitQSemN #-}
